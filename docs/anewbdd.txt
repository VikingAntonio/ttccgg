CREATE TABLE IF NOT EXISTS public.viking_data (
    id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    name TEXT NOT NULL,
    image_url TEXT,
    expansion TEXT,
    rarity TEXT,
    tcg TEXT,
    price TEXT,
    type TEXT, -- 'card', 'product', 'preorder'
    user_id UUID,
    created_at TIMESTAMPTZ DEFAULT NOW()
);

-- RLS Policies
ALTER TABLE public.viking_data ENABLE ROW LEVEL SECURITY;

CREATE POLICY "Allow public read access" ON public.viking_data
    FOR SELECT USING (true);

CREATE POLICY "Allow authenticated insert" ON public.viking_data
    FOR INSERT WITH CHECK (auth.role() = 'authenticated');
